#include <stdio.h>

int main() {
    int a, b;
    
    printf("Enter two integers: ");
    scanf("%d %d", &a, &b);

    if (a > b) {
        printf("Largest number is: %d\n", a);
        printf("Smallest number is: %d\n", b);
    }
    if (b > a) {
        printf("Largest number is: %d\n", b);
        printf("Smallest number is: %d\n", a);
    }
    if (a == b) {
        printf("Both numbers are equal: %d\n", a);
    }

    return 0;
}
