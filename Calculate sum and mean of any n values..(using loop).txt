#include <stdlio.h>

int main() {
    int n, i;
    float sum = 0, mean;

    printf("Enter the number of values (n): ");
    scanf("%d", &n);

    float *values = (float *)malloc(n * sizeof(float));
    if (values == NULL) {
        printf("Memory allocation failed.\n");
        return 1;
    }

    printf("Enter %d values:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%f", &values[i]);
        sum += values[i];
    }

    mean = sum / n;
    
    printf("Sum = %.2f\n", sum);
    printf("Mean = %.2f\n", mean);

    free(values);

    return 0;
}
