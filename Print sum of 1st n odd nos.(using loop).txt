#include <stdio.h>

int main() {
    int n, count = 0, num = 1;
    int sum = 0;

    printf("Enter the value of n: ");
    scanf("%d", &n);

    while (count < n) {
        sum += num;
        num += 2;  // next odd number
        count++;
    }

    printf("Sum of first %d odd numbers = %d\n", n, sum);

    return 0;
}
