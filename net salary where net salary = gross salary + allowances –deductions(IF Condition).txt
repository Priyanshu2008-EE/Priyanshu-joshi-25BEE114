#include <stdio.h>

int main() {
    float gross_salary, allowance = 0, deduction = 0, net_salary;

    printf("Enter the Gross Salary: ");
    scanf("%f", &gross_salary);

    if (gross_salary > 10000) {
        allowance = 0.10 * gross_salary;
        deduction = 0.03 * gross_salary;
    }

    if (gross_salary > 5000 && gross_salary <= 10000) {
        allowance = 0.07 * gross_salary;
        deduction = 0.02 * gross_salary;
    }

    net_salary = gross_salary + allowance - deduction;

    printf("Net Salary = %.2f\n", net_salary);

    return 0;
}
